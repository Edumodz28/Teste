local Lib = loadstring(game:HttpGet("https://raw.githubusercontent.com/7yhx/kwargs_Ui_Library/main/source.lua"))()
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local UI = Lib:Create{
	Theme = "Dark", -- or any other theme
	Size = UDim2.new(0, 1, 0, 1) -- default
}



local function teleportAndAimbot(targetPlayer)
    local player = game.Players.LocalPlayer
    local camera = workspace.CurrentCamera
    local mouse = player:GetMouse()

    local function trackTarget()
        local targetCharacter = targetPlayer.Character
        local targetHead = targetCharacter and targetCharacter:FindFirstChild("Head")

        if targetHead then
            local screenPosition, onScreen = camera:WorldToScreenPoint(targetHead.Position)
            
            if onScreen then
                mouse.Target = targetHead
            end
        end
    end

    local function teleport()
        local playerCharacter = player.Character
        local playerHumanoidRootPart = playerCharacter and playerCharacter:FindFirstChild("HumanoidRootPart")

        if targetPlayer and playerHumanoidRootPart then
            playerHumanoidRootPart.CFrame = targetPlayer.Character:WaitForChild("HumanoidRootPart").CFrame * CFrame.new(0, 0, -5)
        end
    end

    UserInputService.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement then
            trackTarget()
        end
    end)

    RunService.RenderStepped:Connect(function()
        trackTarget()
    end)

    -- Ativar o teleporte e aimbot
    teleport()
end

-- Exemplo de uso (ap칩s o teleporte)
-- Substitua "targetPlayer" pela vari치vel que representa o jogador alvo ap칩s o teleporte
teleportAndAimbot(targetPlayer)
