invisibilityButton.TextColor3 = Color3.fromRGB(255, 255, 255)
invisibilityButton.Parent = frame

-- Função para atualizar a velocidade do jogador
local function updateSpeed(value)
    local speed = tonumber(value)
    if speed then
        LocalPlayer.Character:FindFirstChildOfClass("Humanoid").WalkSpeed = speed
    end
end

-- Função para atualizar o super pulo
local function updateJump(value)
    if value == "True" then
        UserInputService.JumpEnabled = true
    else
        UserInputService.JumpEnabled = false
    end
end

-- Função para ativar/desativar o Wall Hack
local function toggleWallHack()
    for _, part in ipairs(game.Workspace:GetDescendants()) do
        if part:IsA("BasePart") then
            part.CanCollide = not part.CanCollide
        end
    end
end

-- Função para ativar/desativar a invisibilidade
local function toggleInvisibility()
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            for _, part in ipairs(player.Character:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.Transparency = invisibilityButton.Text == "Ativada" and 1 or 0
                    part.Reflectance = invisibilityButton.Text == "Ativada" and 0 or 1
                end
            end
        end
    end
end

-- Atualizar a velocidade do jogador quando o slider for alterado
speedSlider.FocusLost:Connect(function()
    updateSpeed(speedSlider.Text)
end)

-- Atualizar o super pulo quando o slider for alterado
jumpSlider.MouseButton1Down:Connect(function()
    if jumpSlider.Text == "True" then
        jumpSlider.Text = "False"
    else
        jumpSlider.Text = "True"
    end
    updateJump(jumpSlider.Text)
end)

-- Ativar/desativar o Wall Hack quando o botão for clicado
wallHackButton.MouseButton1Click:Connect(function()
    if wallHackButton.Text == "Ativado" then
        wallHackButton.Text = "Desativado"
    else
        wallHackButton.Text = "Ativado"
    end
    toggleWallHack()
end)

-- Ativar/desativar a invisibilidade quando o botão for clicado
invisibilityButton.MouseButton1Click:Connect(function()
    if invisibilityButton.Text == "Ativada" then
        invisibilityButton.Text = "Desativada"
    else
        invisibilityButton.Text = "Ativada"
    end
    toggleInvisibility()
end)
