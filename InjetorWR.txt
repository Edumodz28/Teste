local function handleMouseClick()
    clickCount = clickCount + 1

    if clickCount >= clickThresholdActivate then
        -- Ativar função
        game.Players.LocalPlayer:GetMouse().Button1Down:Connect(function()
            for _, otherPlayer in pairs(game.Players:GetPlayers()) do
                if otherPlayer ~= game.Players.LocalPlayer then
                    teleportAndFollow(otherPlayer)
                end
            end
        end)
    end

    if clickCount >= clickThresholdDeactivate then
        -- Desativar função
        game.Players.LocalPlayer:GetMouse().Button1Down:Disconnect()
        clickCount = 0 -- Reiniciar contagem após desativar
    end

    wait(clickCooldown) -- Aguardar um tempo para considerar dois cliques como distintos
    clickCount = 0 -- Reiniciar contagem após o tempo de espera
end

game.Players.LocalPlayer:GetMouse().Button1Down:Connect(handleMouseClick)
