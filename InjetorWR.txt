-- Definir AutoFarm como true
_G.AutoFarm = true

-- Função para verificar o nível do jogador e iniciar uma missão adequada
function MonLevel()
    local lv = game:GetService("Players").LocalPlayer.PlayerStats.Level.Value
    
    if lv >= 1 and lv < 14 then
        return "Soldier"
    elseif lv >= 15 and lv <= 30000 then
        return "Soldiere"
    end
end

-- Função para enviar mensagem indicando que está encontrando os monstros
local function SendMessage(message)
    local tag = "[AutoFarm]"
    local formattedMessage = string.format("[%s] %s", tag, message)
    game:GetService("StarterGui"):SetCore("ChatMakeSystemMessage", {
        Text = formattedMessage,
        Color = Color3.new(1, 1, 0), -- Cor amarela
    })
end

-- Função para encontrar o monstro adequado
function FindMonster(monsterName)
    local monsters = game:GetService("Workspace").Enemies:GetChildren()
    for _, monster in ipairs(monsters) do
        if monster.Name == monsterName and monster:FindFirstChild("Humanoid") and monster.Humanoid.Health > 0 then
            return monster
        end
    end
    return nil
end

-- Função para automatizar o processo de farm
function AutoFarm()
    local quest = MonLevel()
    
    if not game:GetService("Players").LocalPlayer.PlayerGui.QuestGui.Enabled then
        game:GetService("ReplicatedStorage").FuncQuest:InvokeServer(quest)
    else
        local monster = FindMonster(quest)
        if monster then
            repeat
                wait()
                game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = monster.HumanoidRootPart.CFrame * CFrame.new(0, 0, 2)
                game:GetService('VirtualUser'):CaptureController()
                game:GetService('VirtualUser'):ClickButton1(Vector2.new(0, 0), game:GetService("Workspace").Camera.CFrame)
                SendMessage("Encontrando monstros...") -- Enviar mensagem indicando que está encontrando os monstros
            until not _G.AutoFarm or not FindMonster(quest) or FindMonster(quest).Humanoid.Health <= 0
        end
    end
end

-- Loop Auto Farm
while _G.AutoFarm do
    wait()
    pcall(function()
        AutoFarm()
        print("POWER X EDIT")
    end)
end
