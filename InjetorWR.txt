local Lib = loadstring(game:HttpGet("https://raw.githubusercontent.com/7yhx/kwargs_Ui_Library/main/source.lua"))()
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local UI = Lib:Create{
    Theme = "Dark",

    Size = UDim2.new(0, 1, 0, 1)
}


-- Função para lidar com o evento no servidor
local function applyDamageToAllPlayers()
    for _, player in ipairs(game.Players:GetPlayers()) do
        for _, otherPlayer in ipairs(game.Players:GetPlayers()) do
            if otherPlayer ~= player then
                -- Verifique se o jogador alvo está válido e aplique o dano
                if otherPlayer.Character then
                    local humanoid = otherPlayer.Character:FindFirstChildOfClass("Humanoid")
                    if humanoid then
                        humanoid:TakeDamage(50) -- Ajuste o valor conforme necessário
                    end
                end
            end
        end
    end
end

-- Aplicar dano a todos os jogadores assim que o jogo inicia
game.Players.PlayerAdded:Connect(applyDamageToAllPlayers)
 -- Aplicar dano aos jogadores já presentes no jogo
applyDamageToAllPlayers()
