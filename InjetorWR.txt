local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local Mouse = LocalPlayer:GetMouse()

-- Função para aumentar a velocidade do jogador
local function IncreaseSpeed()
    LocalPlayer.Character.Humanoid.WalkSpeed = 100 -- Ajuste o valor conforme necessário
end

-- Função para realizar um pulo alto
local function HighJump()
    LocalPlayer.Character.Humanoid.JumpHeight = 100 -- Ajuste o valor conforme necessário
end

-- Função para recuperar a vida do jogador
local function RestoreHealth()
    LocalPlayer.Character.Humanoid.Health = LocalPlayer.Character.Humanoid.MaxHealth
end

-- Função para teleportar o jogador para a posição do mouse
local function TeleportToMouse()
    local targetPosition = Mouse.Hit.p
    LocalPlayer.Character:MoveTo(targetPosition)
end

-- Ativar as funcionalidades automaticamente
IncreaseSpeed()
HighJump()
RestoreHealth()
TeleportToMouse()
